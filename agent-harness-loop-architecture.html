async function agentLoop(messages: MessageParam[]) {
  while (true) {
    const res = await client.messages.create({
      model: MODEL, system: SYSTEM, messages, tools: TOOLS,
    });
    messages.push({ role: "assistant", content: res.content });

    switch (res.stop_reason) {
      case "end_turn":    return;                          // 正常结束
      case "pause_turn":  continue;                        // 长任务暂停，回填后续
      case "max_tokens":
      case "stop_sequence":
      case "refusal":     return handleTerminal(res);
    }
    // stop_reason === "tool_use" — 同轮多个 tool_use 可并发执行
    const results = await Promise.all(
      res.content.filter(b => b.type === "tool_use").map(runTool)
    );
    messages.push({ role: "user", content: results });
  }
}

  [H] observability─→  runtime  # 通过事件订阅介入其他面，不反向 import
  [rt] runtime     ─→  ∅ （叶子 · 无内部依赖）

PART · 04 · APPLICATION

base harness 在业务系统中的应用

Harness 是壳，业务是肉 —— 把领域能力通过工具暴露给模型，而不是把模型塞进业务代码里。

① 应用路径 — 把 harness 套到业务上
薄壳厚业务 · Tool Adapter 是唯一耦合面
A · 保持薄壳
业务逻辑不入 harness
harness 只做循环 + 上下文管理 + 工具调用；业务实现继续留在原 service / API。
B · 场景先收窄
选一个有边界的子场景
「客服查单 + 改单」「运维诊断」这类闭环场景先做透 — 不要一上来就做全能助理。
C · Tool Adapter 层
原 API 包一层暴露给模型
不动业务 service，写薄适配把现有 API 翻译成 tool description，独立部署独立演进。
D · 领域知识外置
知识载体随 runtime 能力变
不要硬编码进 harness 或 tool 实现 — 分 C 端 / B 端 / CLI 三档，详见下方专题。
⋯ 知识载体三档 — 跟着 runtime 能力走
D 项展开 · 能跑代码就让知识带代码，不能跑就让知识当资料
TIER · 1 · C 端受限
system prompt + RAG doc
聊天框 / H5 widget / 商品页助手 — 没 shell、没 fs，工具是窄而稳的几个 API。
知识只能内嵌进 prompt 或走 RAG 检索；模型扮演「读资料的人」。
TIER · 2 · B 端工作台
playbook 化的结构 doc
坐席台 / 运维控制台 — 工具丰富，但安全上禁 shell。
markdown SOP 写明「先 A · 再判 B · 最后 C」，本质是结构化提示词包。多数 B 端 agent 该走这档，最被低估。
TIER · 3 · 可运行 CLI
skill — md + script + ref
Claude Code 类 / 内部研发 agent — 有 shell、fs、可装包。
SKILL.md（何时用）+ 可执行脚本（怎么做）+ 引用文件（按需加载） — 渐进披露，把「怎么做」固化进代码。
CORE INSIGHT doc 是「告诉模型怎么做」，skill 是「替模型做掉」。前者是信息，后者是固化的过程 — 能用 skill 就别用 doc，可执行性带来确定性。
② 最小工具集 — 5 类覆盖 80% 场景
先压到 10 个以内 · 工具越多模型选择质量越差
TOOL · 01
search / list
发现实体 — 按条件检索订单、用户、单据，输出 id 列表给后续 get 用。
TOOL · 02
get
读详情 — 拿到 id 之后取完整对象，是模型做判断的主要数据源。
TOOL · 03
act / mutate
执行写操作 — 改单、退款、发通知等，必须带二次确认或权限拦截。
TOOL · 04
ask_user
信息不足反问用户 — 防止模型乱猜参数。优先级高于继续 search。
TOOL · 05
finish / handoff
明确结束或转人工 — 让模型知道有体面的退出路径，避免死循环。
OPT · +1
lookup_doc
RAG 检索业务知识库 / SOP / 政策文档，配上述 5 类基本完备。
RULE
先少后多
别一上来把 30 个内部接口全做成工具，先 5–7 个跑通再加。
RULE
粒度对齐意图
一次调用解决一个用户意图，而不是让模型自己拼 3 个底层 API。
③ 工具的标准 — 区别于传统 API
调用者从「程序员」变成「模型」，所有设计都要为此重做
DIM · 调用者
程序员 → 模型
传统 API 给程序员看，模型工具给 LLM 看 — description 是核心，不是签名。
DIM · 描述
类型签名 → 自然语言
description 必须包含「何时该用 / 何时不该用」，而不是只是参数说明。
DIM · 粒度
细 → 粗
传统 API 靠代码组合；模型工具一次调用对齐一个用户意图。
DIM · 错误
异常码 → 可读文本
错误信息要告诉模型「怎么纠正」，而不是只扔一个 4xx 或堆栈。
DIM · 状态
常带 session → 尽量无状态
每次调用自包含；上下文压缩时仍能被复述、重放、单独审计。
DIM · 返回
结构化 JSON → 可读文本
结构化也行，但要模型能直接理解；不要返回内部枚举码或对象引用。
DIM · 幂等
可选 → 强烈建议
模型会重试、会改主意；写操作没有幂等键，会留下脏数据。
DIM · 边界
上层兜底 → 工具内强校验
不能假设「上层会过滤」— 模型直连，每个 mutate 工具自检权限 / 审批 / 审计。
✓ 核心判据 — 一句话验收
把这条贴在工具评审会议室门口
RULE OF THUMB 把 description 写完后，一个从未看过代码的人能不能判断「什么时候该调这个工具」？
不能 → 说明粒度和语义还没对齐 LLM，回炉重做。不是模型不会用，是工具没写好。
